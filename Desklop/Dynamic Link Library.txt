#include <iostream>
#include <stdexcept>


using namespace std;


class MathLibrary {
public:
    // Method for addition
    static int add(int a, int b) {
        return a + b;
    }


    // Method for subtraction
    static int subtract(int a, int b) {
        return a - b;
    }


    // Method for multiplication
    static int multiply(int a, int b) {
        return a * b;
    }


    // Method for division
    static double divide(int a, int b) {
        if (b == 0) {
            throw invalid_argument("Division by zero is not allowed.");
        }
        return static_cast<double>(a) / b;
    }
};


int main() {
    int num1, num2, choice;


    cout << "Simple Math Application" << endl;
    cout << "Enter two numbers:" << endl;


    cout << "Number 1: ";
    cin >> num1;


    cout << "Number 2: ";
    cin >> num2;


    cout << "\nSelect an operation:" << endl;
    cout << "1. Add" << endl;
    cout << "2. Subtract" << endl;
    cout << "3. Multiply" << endl;
    cout << "4. Divide" << endl;
    cout << "Choice: ";
    cin >> choice;
________________


    try {
        switch (choice) {
            case 1:
                cout << "Result: " << MathLibrary::add(num1, num2) << endl;
                break;
            case 2:
                cout << "Result: " << MathLibrary::subtract(num1, num2) << endl;
                break;
            case 3:
                cout << "Result: " << MathLibrary::multiply(num1, num2) << endl;
                break;
            case 4:
                cout << "Result: " << MathLibrary::divide(num1, num2) << endl;
                break;
            default:
                cout << "Invalid choice!" << endl;
                break;
        }
    } catch (const invalid_argument& e) {
        cout << e.what() << endl;
    }


    return 0;
}
